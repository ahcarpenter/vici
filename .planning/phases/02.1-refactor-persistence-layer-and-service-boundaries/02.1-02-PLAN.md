    ```python
    from src.database import get_sessionmaker  # get_sessionmaker lives in src.database, not src.config — confirmed via grep
    from src.extraction.service import ExtractionService
    from src.extraction.schemas import ExtractionResult
    from src.jobs.repository import JobRepository
    from src.jobs.schemas import JobCreate
    from src.work_requests.repository import WorkRequestRepository
    from src.work_requests.schemas import WorkRequestCreate
    from src.sms.audit_repository import AuditLogRepository
    from src.sms.repository import MessageRepository

    class PipelineOrchestrator:
        def __init__(self, extraction_service, job_repo, work_request_repo,
                     message_repo, audit_repo, pinecone_client, twilio_client):
            ...

        async def run(self, session, sms_text, phone_hash, message_id, user_id,
                      message_sid, from_number) -> ExtractionResult:
            # 1. Classify via GPT (no session)
            result = await self._extraction_service.process(sms_text, phone_hash)
            audit_sid = message_sid

            # 2. Log classification
            await self._audit_repo.write(session, audit_sid, "gpt_classified",
                                         detail=json.dumps({"message_type": result.message_type}),
                                         message_id=message_id)

            # 3. Branch by message_type — invariant: all repo writes and audit writes
            #    happen BEFORE the single session.commit() on each branch.
            if result.message_type == "job_posting" and result.job:
                job = await self._job_repo.create(session, JobCreate(...))
                await session.execute(sa_text("UPDATE message SET message_type = 'job_posting' WHERE id = :mid"), {"mid": message_id})
                await self._audit_repo.write(session, audit_sid, "job_created",
                                             detail=json.dumps({"job_id": job.id}),
                                             message_id=message_id)
                await session.commit()  # Single commit for this branch

                # Fire-and-forget Pinecone (after commit — failure should not roll back job)
                try:
                    await self._pinecone_client(job_id=job.id, description=result.job.description,
                                                phone_hash=phone_hash, openai_client=self._extraction_service._client,
                                                settings=self._extraction_service._settings)
                except Exception as e:
                    log.error("pinecone_write_failed", job_id=job.id, error=str(e))
                    async with get_sessionmaker()() as s2:  # separate session for retry queue
                        await s2.execute(sa_text("INSERT INTO pinecone_sync_queue ..."), ...)
                        await s2.commit()

            elif result.message_type == "worker_goal" and result.worker:
                wr = await self._work_request_repo.create(session, WorkRequestCreate(...))
                await session.execute(sa_text("UPDATE message SET message_type = 'worker_goal' WHERE id = :mid"), {"mid": message_id})
                await self._audit_repo.write(session, audit_sid, "work_request_created",
                                             detail=json.dumps({"work_request_id": wr.id}),
                                             message_id=message_id)
                await session.commit()  # Single commit for this branch — audit write is BEFORE commit

            else:  # unknown
                await session.execute(sa_text("UPDATE message SET message_type = 'unknown' WHERE id = :mid"), {"mid": message_id})
                await session.commit()

            return result
    ```
