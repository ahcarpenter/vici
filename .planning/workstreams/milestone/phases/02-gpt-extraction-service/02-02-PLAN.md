    ```python
    from pinecone import PineconeAsyncio, Vector
    from src.extraction.constants import EMBEDDING_MODEL, EMBEDDING_DIMS

    async def write_job_embedding(
        job_id: int,
        description: str,
        phone_hash: str,
        openai_client,
        settings,
    ) -> None:
        """Generate embedding via OpenAI and upsert to Pinecone. Raises on any failure."""
        emb_resp = await openai_client.embeddings.create(
            model=EMBEDDING_MODEL,
            input=description,
        )
        vector = emb_resp.data[0].embedding  # list[float], EMBEDDING_DIMS elements

        async with PineconeAsyncio(api_key=settings.pinecone_api_key) as pc:
            async with pc.IndexAsyncio(host=settings.pinecone_index_host) as idx:
                await idx.upsert(vectors=[
                    Vector(
                        id=str(job_id),
                        values=vector,
                        metadata={"phone_hash": phone_hash},
                    )
                ])
    ```

    ```python
    async def process(
        self,
        sms_text: str,
        phone_hash: str,
        message_id: int,
        session: AsyncSession,
    ) -> ExtractionResult:
        from datetime import date
        from sqlalchemy import text as sa_text
        from src.jobs.repository import JobRepository
        from src.jobs.schemas import JobCreate
        from src.workers.repository import WorkerRepository
        from src.workers.schemas import WorkerCreate
        from src.extraction.pinecone_client import write_job_embedding

        user_message = f"Today is {date.today().isoformat()}. Message: {sms_text}"
        result = await self._call_with_retry(user_message)

        log.info("gpt_classified", message_type=result.message_type, phone_hash=phone_hash)
        await self._write_audit(session, message_id, "gpt_classified",
                                {"message_type": result.message_type})

        if result.message_type == "job_posting" and result.job:
            job_create = JobCreate(
                phone_hash=phone_hash,
                description=result.job.description,
                location=result.job.location,
                pay_rate=result.job.pay_rate,
                pay_type=result.job.pay_type,
                estimated_duration_hours=result.job.estimated_duration_hours,
                raw_duration_text=result.job.raw_duration_text,
                ideal_datetime=result.job.ideal_datetime,
                raw_datetime_text=result.job.raw_datetime_text,
                inferred_timezone=result.job.inferred_timezone,
                datetime_flexible=result.job.datetime_flexible,
                raw_sms=sms_text,
            )
            # Single transaction: create job + update message_type
            job = await JobRepository.create(session, job_create)
            await session.execute(
                sa_text("UPDATE inbound_message SET message_type = 'job_posting' WHERE id = :mid"),
                {"mid": message_id},
            )
            await session.commit()
            await self._write_audit(session, message_id, "job_created", {"job_id": job.id})

            # Fire-and-forget Pinecone upsert
            try:
                await write_job_embedding(
                    job_id=job.id,
                    description=result.job.description,
                    phone_hash=phone_hash,
                    openai_client=self._client,
                    settings=self._settings,
                )
            except Exception as e:
                log.error("pinecone_write_failed", job_id=job.id, error=str(e))
                await self._write_audit(session, message_id, "pinecone_write_failed",
                                        {"job_id": job.id, "error": str(e)})
                await self._enqueue_pinecone_sync(session, job.id)

        elif result.message_type == "worker_goal" and result.worker:
            worker_create = WorkerCreate(
                phone_hash=phone_hash,
                target_earnings=result.worker.target_earnings,
                target_timeframe=result.worker.target_timeframe,
                raw_sms=sms_text,
            )
            worker = await WorkerRepository.create(session, worker_create)
            await session.execute(
                sa_text("UPDATE inbound_message SET message_type = 'worker_goal' WHERE id = :mid"),
                {"mid": message_id},
            )
            await session.commit()
            await self._write_audit(session, message_id, "work_request_created",
                                    {"work_request_id": worker.id})

        else:  # unknown or GPT refusal
            await session.execute(
                sa_text("UPDATE inbound_message SET message_type = 'unknown' WHERE id = :mid"),
                {"mid": message_id},
            )
            await session.commit()

        return result

    async def _write_audit(self, session, message_id: int, event: str, detail: dict):
        import json
        from datetime import datetime
        from sqlalchemy import text as sa_text
        await session.execute(
            sa_text(
                "INSERT INTO audit_log (message_sid, event, detail, created_at) "
                "VALUES (:sid, :event, :detail, :created_at)"
            ),
            {
                "sid": str(message_id),
                "event": event,
                "detail": json.dumps(detail),
                "created_at": datetime.utcnow(),
            },
        )
        await session.commit()

    async def _enqueue_pinecone_sync(self, session, job_id: int):
        from datetime import datetime
        from sqlalchemy import text as sa_text
        await session.execute(
            sa_text(
                "INSERT INTO pinecone_sync_queue (job_id, status, attempts, created_at) "
                "VALUES (:job_id, 'pending', 0, :created_at)"
            ),
            {"job_id": job_id, "created_at": datetime.utcnow()},
        )
        await session.commit()
    ```
